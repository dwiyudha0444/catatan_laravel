1. composer require midtrans/midtrans-php
2. pada .env

# API KEY MIDTRANS
MIDTRANS_MERCHANT_ID=G304392264
MIDTRANS_CLIENT_KEY=SB-Mid-client-PhxGO1UuxaFXR6CC
MIDTRANS_SERVER_KEY=SB-Mid-server-jLq5REMnYGqlarUbKaS_D_C4

3. tambahkan config midtrans.php

<?php

return [

    'merchant_id' => env('MIDTRANS_MERCHANT_ID'),
    'client_key' => env('MIDTRANS_CLIENT_KEY'),
    'server_key' => env('MIDTRANS_SERVER_KEY'),

];

4. buat model dan controller

php artisan make:model Order -mc
php artisan make:controller client/OrderController --resource
